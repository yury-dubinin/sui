Represents a module in Move, a library that defines struct types
and functions that operate on these types.


```graphql
type MoveModule {
  package: MovePackage!
  name: String!
  fileFormatVersion: Int!
  friendConnection(
    first: Int
    after: String
    last: Int
    before: String
  ): MoveModuleConnection!
  struct(
    name: String!
  ): MoveStruct
  structConnection(
    first: Int
    after: String
    last: Int
    before: String
  ): MoveStructConnection
  function(
    name: String!
  ): MoveFunction
  functionConnection(
    first: Int
    after: String
    last: Int
    before: String
  ): MoveFunctionConnection
  bytes: Base64
  disassembly: String
}
```
